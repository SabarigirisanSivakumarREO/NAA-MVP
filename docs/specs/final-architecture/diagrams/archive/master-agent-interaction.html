Browser Agent → Analysis Agent
interface PageExecutionResult {
  url: string;
  template_id: string;
  page_type: PageType;
  states: StateNode[];                // one or more meaningful states
  multi_state_perception: MultiStatePerception;
  interaction_log: InteractionRecord[];
  cost_spent_usd: number;
  degraded: boolean;                  // true if Tier C fallback fired
}

State Exploration → Analysis Agent
interface MultiStatePerception {
  page_url: string;
  default_state: AnalyzePerception;   // always present
  hidden_states: Array<{              // 0-14 additional states
    state_id: StateId;
    interaction_path: InteractionPath;
    perception: AnalyzePerception;
  }>;
  merged_view: AnalyzePerception;     // cross-state union for evaluation
  state_provenance: Record<string, StateId>;  // element → source state
}
